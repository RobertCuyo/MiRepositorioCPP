<========================factorial========================>
#include <iostream>
using namespace std;
int factorial(int num) {/*una funcion que retorna el resultado atraves de un operador condicional
	la recursividad sigue mientras q num sea mayor a 0*/
	return (num == 0) ? 1 : num * factorial(num - 1);//operador condicional
}
int main() {
	int numero=0; cin >> numero;
	cout << factorial(numero) << endl;// se llama a la funcion recursiva
	return 0;
}
<========================fibonacci========================>
#include <iostream>
using namespace std;
int fibonacci(int num) {
	return (num < 2) ? num : fibonacci(num - 1) + fibonacci(num - 2);
	/*se retorna el valor del operador condicional */
}
int main() {
	for (int i = 0; i <= 7; i++) {
		cout <<"  "<< fibonacci(i) ;
	}
	return 0;
}
<========================Sumatoria de 1+2+3+4+5+...n ========================>
#include <iostream>
using namespace std;
int main() {
	int aux=0;
	for (int i = 1; i <= 5; i++) {
		if (i<5) {
			cout << i<<" + " ;
		}
		else {cout << i;}
		aux += i;// aux se en encarga de guardar la suma de los i
	}
	cout << " = " << aux; // se muestra la suma final 
	return 0;
}

<========================Sumatoria de cuadrados ========================>
#include <iostream>
using namespace std;
int main() {
	int aux = 0,num=0;//inicializa el valor de las variables con sus valores por defecto
	cout << "Ingrese un numero: "; cin >> num;
	for (int i = 1; i <= num; i++) {
		aux += (i * i); // aux suma los cuadros de los i 
		if (i < num) {
			cout << i << "^2 + ";// imprime i^2 + para tener un acabado 
		}
		else { cout << i << "^2" << " = " << aux; }	muestra el resultado de las sumas del cuadrado de i
	}
	return 0;
}
